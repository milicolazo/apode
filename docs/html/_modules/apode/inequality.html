class InequalityMeasures:
    """Inequality measures for Apode.

    The following inequality measures are implemented:
    - gini: Gini Index
    - entropy: Generalized Entropy Index
    - atkinson: Atkinson Index
    - rrange: Relative Range
    - rad: Relative average deviation
    - cv: Coefficient of variation
    - sdlog: Standard deviation of log
    - merhan: Merhan index
    - piesch: Piesch Index
    - bonferroni: Bonferroni Indices
    - kolm: Kolm Index
    - ratio: ratios
    - Hagenaars Index
    - Chakravarty Indices

    Parameters
    ----------
    method : String
        Inequality measure.
    **kwargs
        Arbitrary keyword arguments.

    """

    idf = attr.ib()

    def __call__(self, method=None, **kwargs):
        """Return the ApodeData object."""
        method = "gini" if method is None else method
        method_func = getattr(self, method)
        return method_func(**kwargs)

[docs]    def entropy(self, alpha=0):
        """General Entropy index.

        The entropy index measures the proportion of the population that
        counted as poor.

        Parameters
        ----------
        alpha: float, optional(default=0)

        Return
        ------
        out: float
            Entropy index measures.

        """
        a = alpha
        y = self.idf.data[self.idf.varx].values
        n = len(y)
        if n == 0:
            return 0
        u = np.mean(y)
        if a == 0.0:
            return np.sum(np.log(u / y)) / n
        elif a == 1.0:
            return np.sum((y / u) * np.log(y / u)) / n
        return (1 / (a * (a - 1))) * (np.sum(pow(y / u, a)) / n - 1)



[docs]    def atkinson(self, alpha=2):
        """Atkinson index.

        The Atkinson index measures the proportion of the population that
        counted as poor.

        Parameters
        ----------
        alpha: float, optional(default=2)

        Return
        ------
        out: float
            Atkinson index measures.

        """
        y = self.idf.data[self.idf.varx].values
        if alpha <= 0:
            raise ValueError("Alpha must be strictly positive (>0.0)")
        n = len(y)
        if n == 0:
            return 0
        if alpha == 1:
            y_nz = y[y != 0]
            ylog = np.log(y_nz)
            h = np.mean(ylog)
            return 1 - np.exp(h) / np.mean(y_nz)
        else:
            with np.errstate(divide='ignore'):
                a1 = np.sum(np.power(y, 1 - alpha)) / n
                return 1 - np.power(a1, 1 / (1 - alpha)) / np.mean(y)

        y = self.idf.data[self.idf.varx].values

[docs]    def gini(self):
        """Gini Coefficient.

        The Gini Coefficient

        Return
        ------
        out: float
            Gini Coefficient.

        """
        y = self.idf.data[self.idf.varx].values
        y = np.sort(y)
        n = len(y)
        if n == 0:
            return 0
        u = np.mean(y)
        a = 0
        for i in range(0, n):
            a = a + (n - i) * y[i]
        g = (n + 1) / (n - 1) - 2 / (n * (n - 1) * u) * a
        g = g * (n - 1) / n
        return g



[docs]    def merhan(self):
        """Merhan Coefficient.

        The Merhan Coefficient

        Return
        ------
        out: float
            Merhan Coefficient.

        """
        y = self.idf.data[self.idf.varx].values
        y = np.sort(y)
        n = len(y)
        if n == 0:
            return 0
        u = np.mean(y)
        f = 1.0 / (n * u)
        syi = y[0]
        pi = 1.0 / n
        qi = f * y[0]
        m = pi - qi
        for i in range(1, n - 1):
            pi = i / n
            syi = syi + y[i]
            qi = f * syi
            m = m + (1 - pi) * (pi - qi)
        return m * 6 / n



[docs]    def piesch(self):
        """Piesch Coefficient.

        The Piesch Coefficient

        Return
        ------
        out: float
            Piesch Coefficient.

        """
        y = self.idf.data[self.idf.varx].values
        y = np.sort(y)
        n = len(y)
        if n == 0:
            return 0
        u = np.mean(y)
        f = 1.0 / (n * u)
        syi = y[0]
        pi = 1.0 / n
        qi = f * y[0]
        m = pi - qi
        for i in range(1, n - 1):
            pi = i / n
            syi = syi + y[i]
            qi = f * syi
            m = m + pi * (pi - qi)
        return m * 3 / n



[docs]    def bonferroni(self):
        """Bonferroni Coefficient.

        The Bonferroni Coefficient

        Return
        ------
        out: float
            Bonferroni Coefficient.

        """
        y = self.idf.data[self.idf.varx].values
        y = np.sort(y)
        n = len(y)
        if n == 0:
            return 0
        s = y[0]
        x = y[0]
        for i in range(1, n - 1):  # i=2,n-1
            x = x + y[i]
            s = s + x / i
        u = (x + y[n - 1]) / n
        return 1 - (1 / ((n - 1) * u)) * s

        n = len(y)
        if n == 0:
            return 0
        u = np.mean(y)
        return (1 / alpha) * (np.log((1.0 / n) *
                                     np.sum(np.exp(alpha * (u - y)))))



[docs]    def ratio(self, alpha):
        """Ratio Coefficient.

        The Ratio Coefficient

        Parameters
        ----------
        alpha: float

        Return
        ------
        out: float
            Ratio Coefficient.

        """
        y = self.idf.data[self.idf.varx].values
        if (alpha < 0) or (alpha > 1):
            raise ValueError(f"'alpha' must be in [0,1]. Found '{alpha}'")
        y = np.sort(y)
        n = len(y)
        if n == 0:
            return 0
        k = int(np.floor(alpha * n))
        return np.mean(y[:k]) / np.mean(y[n - k:])
